2. 필수 코드 스니펫 (GSAP + ScrollTrigger)
js
Copy
Edit
gsap.registerPlugin(ScrollTrigger);

const wrap   = document.querySelector(".videoWrap");
const rect   = wrap.getBoundingClientRect();
const vw     = window.innerWidth;
const vh     = window.innerHeight;
const scale  = Math.max(vw / rect.width, vh / rect.height);

// 중심으로 보낼 x·y 이동 값
const x = vw * 0.5 - (rect.left + rect.width  * 0.5);
const y = vh * 0.5 - (rect.top  + rect.height * 0.5);

gsap.timeline({
  scrollTrigger:{
    trigger:".hero",          // 텍스트 블록
    start:"top center",
    end:"+=200%",
    scrub:true,
    pin:true,
    anticipatePin:1
  }
})
.to(wrap, {              // ① 위치 보정
  x, y,
  ease:"none"
})
.to(wrap, {              // ② 스케일
  scale,
  transformOrigin:"center center",
  ease:"power1.out"
}, "<");                 // 동시에 시작
왜 x,y 를 먼저 주나?
이동값·스케일을 동시에 적용하면 확대 과정 동안 항상 중앙을 향해 커집니다. 이동 없이 스케일만 주면 원래 좌표 기준으로 확대되어 “한쪽으로 치우쳐” 보여요. 
gsap.com

anticipatePin 은 첫 핀 순간 튀는 현상을 줄여 줍니다. 
gsap.com

4. 바닐라 JS 로직 요약 (Intersection Observer ver.)
observer.observe(triggerEl) → 교차 시 fixed-layer 클래스를 videoWrap 에 부여

이 클래스에서 position:fixed; top:50%; left:50%; translate:-50% -50% 지정

동시에 scale 을 CSS 변수로 두고 scrollY 대비 비례 증가(예 calc(1 + var(--progress)*4) )

뷰포트 끝나면 클래스 제거, 다음 섹션 자연 노출
